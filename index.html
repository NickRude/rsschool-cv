#include <stdio.h>
#include <unistd.h>

int ft_strncmp(char *s1, char *s2, unsigned int n)
{
    unsigned int i;
    i = 0;

    while (i < n && s1[i] != '\0' && s2[i] != '\0' && s1[i] == s2[i])
    {
        i++;
    }

    if (i == n)
        return 0;

    else
        return s1[i] - s2[i];
}

int main(void)
{
    char s1[] = "aompare the text";
    char s2[] = "bompare the text";
    unsigned int a = 5;
    int result = ft_strncmp(s1, s2, a);
    if (result == 0)
        printf("Text is the same");
    if (result > 0)
        printf("Result is positive");
    if (result < 0)
        printf("Result is negative");
}
